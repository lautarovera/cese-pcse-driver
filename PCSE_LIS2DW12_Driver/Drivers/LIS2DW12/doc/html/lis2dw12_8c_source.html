179 static lis2dw12_status_t get_raw_fifo(int16_t *raw_x, int16_t *raw_y, int16_t *raw_z, uint8_t samples)
180 {
181  lis2dw12_status_t status = LIS2DW12_NOK;
182  /* Each sample is composed by 6 bytes, since there are three axis and each axis is defined by the high and low byte registers */
183  uint8_t size = 6u;
184  uint8_t data[samples][size];
185 
186  if (NULL != raw_x && NULL != raw_y && NULL != raw_z) {
187  for (uint8_t i = 0u; i < samples; i++) {
188  i2c_status_t i2c_status = lis2dw12.i2c_mem_read(LIS2DW12_SAD_R, LIS2DW12_REG_OUT_X_L, LIS2DW12_REG_SIZE, &data[i][0u], size);
189 
190  if (I2C_OK == i2c_status) {
191  *raw_x++ = (int16_t)((uint16_t)data[i][1u] << 8u | (uint16_t)data[i][0u]);
192  *raw_y++ = (int16_t)((uint16_t)data[i][3u] << 8u | (uint16_t)data[i][2u]);
193  *raw_z++ = (int16_t)((uint16_t)data[i][5u] << 8u | (uint16_t)data[i][4u]);
194 
195  status = LIS2DW12_OK;
196  }
197  }
198  }
199 
200  return status;
201 }

203 static int16_t raw_to_mg(int16_t raw_data)
204 {
205  float sen_fac = 0;
206  uint8_t bit_shift = (lis2dw12.config.lp_mode == LIS2DW12_LP_MODE_1) ? 4u : 2u;
207 
208  raw_data >>= bit_shift;
209 
210  switch (lis2dw12.config.fs)
211  {
212  case LIS2DW12_FS_2_G:
213  sen_fac = (lis2dw12.config.lp_mode == LIS2DW12_LP_MODE_1) ? 0.976 : 0.244;
214  break;
215  case LIS2DW12_FS_4_G:
216  sen_fac = (lis2dw12.config.lp_mode == LIS2DW12_LP_MODE_1) ? 1.952 : 0.488;
217  break;
218  case LIS2DW12_FS_8_G:
219  sen_fac = (lis2dw12.config.lp_mode == LIS2DW12_LP_MODE_1) ? 3.904 : 0.976;
220  break;
221  case LIS2DW12_FS_16_G:
222  sen_fac = (lis2dw12.config.lp_mode == LIS2DW12_LP_MODE_1) ? 7.808 : 1.952;
223  break;
224  default:
225  /* Data will be 0 */
226  break;
227  }
228 
229  raw_data *= sen_fac;
230 
231  return raw_data;
232 }

235 lis2dw12_status_t lis2dw12_init(lis2dw12_config_t usr_config)
236 {
237  lis2dw12_status_t status = LIS2DW12_OK;
238  uint8_t reg_ctrl1_val = 0u;
239  uint8_t reg_ctrl2_val = 0xC;
240  uint8_t reg_ctrl6_val = 0u;
241  uint8_t reg_fifo_ctrl_val = 0u;
242 
243  #ifdef STM32F429xx
244  lis2dw12.i2c_init = stm32f429xx_i2c_init;
245  lis2dw12.i2c_master_tx = stm32f429xx_i2c_master_tx;
246  lis2dw12.i2c_master_rx = stm32f429xx_i2c_master_rx;
247  lis2dw12.i2c_mem_write = stm32f429xx_i2c_mem_write;
248  lis2dw12.i2c_mem_read = stm32f429xx_i2c_mem_read;
249  lis2dw12.i2c_is_dev_ready = stm32f429xx_i2c_is_dev_ready;
250  #endif
251 
252  lis2dw12.config.odr = (usr_config.odr > LIS2DW12_ODR_1600_HZ) ? LIS2DW12_ODR_DEFAULT : usr_config.odr;
253  lis2dw12.config.mode = (usr_config.mode > LIS2DW12_MODE_ON_DM) ? LIS2DW12_MODE_DEFAULT : usr_config.mode;
254  lis2dw12.config.lp_mode = (usr_config.lp_mode > LIS2DW12_LP_MODE_4) ? LIS2DW12_LP_MODE_DEFAULT : usr_config.lp_mode;
255  lis2dw12.config.bw_filt = (usr_config.bw_filt > LIS2DW12_BW_FILT_4) ? LIS2DW12_BW_FILT_DEFAULT : usr_config.bw_filt;
256  lis2dw12.config.fs = (usr_config.fs > LIS2DW12_FS_16_G) ? LIS2DW12_FS_DEFAULT : usr_config.fs;
257  lis2dw12.config.fds = (usr_config.fds == LIS2DW12_FDS_HP_FILT) ? usr_config.fds : LIS2DW12_FDS_DEFAULT;
258  lis2dw12.config.low_noise = (usr_config.low_noise == LIS2DW12_LOW_NOISE_ON) ? usr_config.low_noise : LIS2DW12_LOW_NOISE_DEFAULT;
259  lis2dw12.config.fifo_mode = (usr_config.fifo_mode == LIS2DW12_FIFO_MODE_ON) ? usr_config.fifo_mode : LIS2DW12_FIFO_MODE_DEFAULT;
260  lis2dw12.config.fifo_ths = usr_config.fifo_ths - 1u;
261  lis2dw12.config.fifo_ths = (usr_config.fifo_ths > 31u) ? 31u : usr_config.fifo_ths;
262 
263  lis2dw12.i2c_init();
264 
265  reg_ctrl1_val = lis2dw12.config.odr << 4u | lis2dw12.config.mode << 2u | lis2dw12.config.lp_mode;
266  reg_ctrl6_val = lis2dw12.config.bw_filt << 6u | lis2dw12.config.fs << 4u | lis2dw12.config.fds << 3u | lis2dw12.config.low_noise << 2u;
267  reg_fifo_ctrl_val = lis2dw12.config.fifo_mode << 5u | lis2dw12.config.fifo_ths;
268 
269  status = set_reg(LIS2DW12_REG_CTRL1, reg_ctrl1_val);
270  status &= set_reg(LIS2DW12_REG_CTRL2, reg_ctrl2_val);
271  status &= set_reg(LIS2DW12_REG_CTRL6, reg_ctrl6_val);
272  status &= set_reg(LIS2DW12_REG_FIFO_CTRL, reg_fifo_ctrl_val);
273 
274  return status;
275 }

383 lis2dw12_status_t lis2dw12_get_fifo_mg(int16_t *buffer_x, int16_t *buffer_y, int16_t *buffer_z, uint8_t buffer_len)
384 {
385  lis2dw12_status_t status = LIS2DW12_NOK;
386  uint8_t samples = lis2dw12.config.fifo_ths + 1u;
387 
388  if (buffer_len >= samples) {
389  uint8_t reg_fifo_ctrl_val = LIS2DW12_FIFO_MODE_OFF << 5u | lis2dw12.config.fifo_ths;
390 
391  if (LIS2DW12_OK == get_raw_fifo(buffer_x, buffer_y, buffer_z, samples)) {
392  int16_t tmp_x = 0;
393  int16_t tmp_y = 0;
394  int16_t tmp_z = 0;
395 
396  while (samples--) {
397  tmp_x = *buffer_x;
398  tmp_y = *buffer_y;
399  tmp_z = *buffer_z;
400 
401  *buffer_x++ = raw_to_mg(tmp_x);
402  *buffer_y++ = raw_to_mg(tmp_y);
403  *buffer_z++ = raw_to_mg(tmp_z);
404  }
405  }
406 
407  /* Enables again the FIFO */
408  status = set_reg(LIS2DW12_REG_FIFO_CTRL, reg_fifo_ctrl_val);
409 
410  reg_fifo_ctrl_val = lis2dw12.config.fifo_mode << 5u | lis2dw12.config.fifo_ths;
411 
412  status &= set_reg(LIS2DW12_REG_FIFO_CTRL, reg_fifo_ctrl_val);
413  }
414 
415  return status;
